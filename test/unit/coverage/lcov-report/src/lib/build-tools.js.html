export function clouds (count = 10, xMin = 0, xMax = 100, yMin = 0, yMax = 100, config = {}) {
  const clouds = []
 
  for (let i = 0; i < count; i++) {
    clouds.push(Object.assign({}, SHAPE, config, {
      x: Math.random() * (xMax - xMin) + xMin,
      y: Math.random() * (yMax - yMin) + yMin,
      w: Math.random() * 100,
      h: 1,
      x0: 0,
      y0: 0,
      dx: Math.random() * -0.1,
      dy: 0
    }))
  }
 
  return clouds
}
